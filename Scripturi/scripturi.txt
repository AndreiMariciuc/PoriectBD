DROP DATABASE IF EXISTS aplicatie;
CREATE DATABASE aplicatie;
USE aplicatie;

DROP TABLE IF EXISTS roluri;
CREATE TABLE roluri (
  id_rol INT NOT NULL AUTO_INCREMENT,
  descriere_rol VARCHAR(16) NOT NULL,
  PRIMARY KEY (id_rol),
  UNIQUE INDEX id_rol_UNIQUE (id_rol ASC) VISIBLE);

DROP TABLE IF EXISTS an_universitar;
CREATE TABLE an_universitar (
  id_an_universitar INT NOT NULL,
  semestru INT NOT NULL,
  data_inceput DATE NOT NULL,
  data_sfarsit DATE NOT NULL,
  an_inceput YEAR NOT NULL,
  PRIMARY KEY (id_an_universitar),
  UNIQUE INDEX id_an_universitar_UNIQUE (id_an_universitar ASC) VISIBLE);

DROP TABLE IF EXISTS perioade;
CREATE TABLE perioade (
  id_perioada INT NOT NULL,
  descriere VARCHAR(32) NOT NULL,
  PRIMARY KEY (id_perioada),
  UNIQUE INDEX id_perioada_UNIQUE (id_perioada ASC) VISIBLE);

DROP TABLE IF EXISTS zile;
CREATE TABLE zile (
  id_zi INT NOT NULL AUTO_INCREMENT,
  zi VARCHAR(16) NOT NULL,
  PRIMARY KEY (id_zi),
  UNIQUE INDEX id_zi_UNIQUE (id_zi ASC) VISIBLE);

  DROP TABLE IF EXISTS activitati;
  CREATE TABLE activitati (
  id_activitate INT UNSIGNED NOT NULL AUTO_INCREMENT,
  denumire_activitate VARCHAR(32) NULL,
  nr_max_participanti INT NULL,
  PRIMARY KEY (id_activitate),
  UNIQUE INDEX id_activitate_UNIQUE (id_activitate ASC) VISIBLE);

DROP TABLE IF EXISTS departamente;
CREATE TABLE departamente (
  id_departament INT NOT NULL,
  cod_departament VARCHAR(8) NOT NULL,
  descriere_departament VARCHAR(32) NOT NULL,
  PRIMARY KEY (id_departament));

DROP TABLE IF EXISTS cursuri;
CREATE TABLE cursuri (
  id_curs INT NOT NULL,
  denumire VARCHAR(32) NOT NULL,
  descriere VARCHAR(128) NOT NULL DEFAULT '',
  PRIMARY KEY (id_curs));

DROP table IF EXISTS users;
CREATE TABLE users (
  id_user INT UNSIGNED NOT NULL AUTO_INCREMENT,
  id_rol INT NULL,
  CNP CHAR(13) NOT NULL,
  nume VARCHAR(64) NOT NULL,
  prenume VARCHAR(64) NOT NULL,
  adresa VARCHAR(128) NULL,
  nr_tel CHAR(10) NULL,
  email VARCHAR(64) NOT NULL,
  IBAN VARCHAR(34) NOT NULL,
  nr_contract INT NULL,
  username VARCHAR(32) NOT NULL,
  parola VARCHAR(32) NOT NULL,
  PRIMARY KEY (id_user),
  UNIQUE INDEX id_user_UNIQUE (id_user ASC) VISIBLE,
  UNIQUE INDEX CNP_UNIQUE (CNP ASC) VISIBLE,
  UNIQUE INDEX username_UNIQUE (username ASC) VISIBLE,
  INDEX id_rol_idx (id_rol ASC) VISIBLE,
  CONSTRAINT id_rol
    FOREIGN KEY (id_rol)
    REFERENCES aplicatie.roluri (id_rol)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);

DROP TABLE IF EXISTS status_activ;
CREATE TABLE status_activ (
  id_status INT NOT NULL,
  Descriere VARCHAR(16) NOT NULL,
  PRIMARY KEY (id_status));

DROP TABLE IF EXISTS studenti;
CREATE TABLE studenti (
  id_student INT UNSIGNED NOT NULL,
  nr_ore INT NOT NULL DEFAULT 6,
  an_studiu INT NOT NULL DEFAULT 1,
  id_status INT NOT NULL DEFAULT 0,
  PRIMARY KEY (id_student),
  UNIQUE INDEX id_student_UNIQUE (id_student ASC) VISIBLE,
  INDEX id_status_activ_idx (id_status ASC) VISIBLE,
  CONSTRAINT id_status_activ
    FOREIGN KEY (id_status)
    REFERENCES aplicatie.status_activ (id_status)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT id_user_stud
    FOREIGN KEY (id_student)
    REFERENCES aplicatie.users (id_user)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);

DROP TABLE IF EXISTS profesori;
CREATE TABLE profesori (
  id_profesor INT UNSIGNED NOT NULL,
  nr_min_ore INT NOT NULL DEFAULT 8,
  nr_max_ore INT NOT NULL DEFAULT 32,
  id_departament INT NULL,
  PRIMARY KEY (id_profesor),
  UNIQUE INDEX id_profesor_UNIQUE (id_profesor ASC) VISIBLE,
  INDEX id_departament_idx (id_departament ASC) VISIBLE,
  CONSTRAINT id_user_prof
    FOREIGN KEY (id_profesor)
    REFERENCES users (id_user)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT id_departament
    FOREIGN KEY (id_departament)
    REFERENCES departamente (id_departament)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);

DROP TABLE IF EXISTS curs_activitati;
CREATE TABLE curs_activitati (
  id_ca INT UNSIGNED NOT NULL AUTO_INCREMENT,
  id_curs INT NOT NULL,
  id_activ INT UNSIGNED NOT NULL,
  perioada INT,
  nr_zi_sapt INT,
  procent INT,
  id_prof_titular INT UNSIGNED,
  id_prof_delegat INT UNSIGNED,
  ora_inceput int,
  durata INT NOT NULL DEFAULT 1,
  nr_max_participanti int,
  PRIMARY KEY (id_ca),
  UNIQUE INDEX id_ca_UNIQUE (id_ca ASC) VISIBLE,
  INDEX id_curs_idx (id_curs ASC) VISIBLE,
  INDEX id_activ_idx (id_activ ASC) VISIBLE,
  INDEX perioada_idx (perioada ASC) VISIBLE,
  INDEX nr_zi_saptamana_idx (nr_zi_sapt ASC) VISIBLE,
  INDEX id_prof_apartinator_idx (id_prof_titular ASC) VISIBLE,
  CONSTRAINT prof_delegat
	FOREIGN KEY (id_prof_delegat)
    REFERENCES aplicatie.profesori (id_profesor)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT id_curs
    FOREIGN KEY (id_curs)
    REFERENCES aplicatie.cursuri (id_curs)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT id_activ
    FOREIGN KEY (id_activ)
    REFERENCES aplicatie.activitati (id_activitate)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT perioada
    FOREIGN KEY (perioada)
    REFERENCES aplicatie.perioade (id_perioada)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT nr_zi_saptamana
    FOREIGN KEY (nr_zi_sapt)
    REFERENCES aplicatie.zile (id_zi)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT id_profesor_titular
    FOREIGN KEY (id_prof_titular)
    REFERENCES aplicatie.profesori (id_profesor)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);

DROP TABLE IF EXISTS studenti_activitati;
CREATE TABLE studenti_activitati (
  id_student INT UNSIGNED NOT NULL,
  id_activitate INT UNSIGNED NOT NULL,
  nota INT,
  data_notare       date,
  INDEX id_activitate_idx (id_activitate ASC) VISIBLE,
  INDEX id_student_idx (id_student ASC) VISIBLE,
  CONSTRAINT id_student
    FOREIGN KEY (id_student)
    REFERENCES aplicatie.studenti (id_student)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT id_activitate
    FOREIGN KEY (id_activitate)
    REFERENCES aplicatie.curs_activitati (id_ca)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);

DROP TABLE IF EXISTS grupe;
CREATE TABLE grupe (
  id_grupa INT NOT NULL,
  id_curs INT NULL,
  id_admin_grupa INT UNSIGNED NOT NULL,
  id_profesor_raspunzator INT UNSIGNED, -- poate fi null deoarece o grupa de aprofundare poate sa nu aiba profesor
  nr_max_particip INT NOT NULL DEFAULT 30,
  PRIMARY KEY (id_grupa),
  INDEX id_curs_idx (id_curs ASC) VISIBLE,
  INDEX id_admin_grupa_idx (id_admin_grupa ASC) VISIBLE,
  CONSTRAINT id_cursG
    FOREIGN KEY (id_curs)
    REFERENCES cursuri (id_curs)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
CONSTRAINT prof_raspunzator
    FOREIGN KEY (id_profesor_raspunzator)
    REFERENCES profesori (id_profesor)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT id_admin_grupa
    FOREIGN KEY (id_admin_grupa)
    REFERENCES studenti (id_student)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);

DROP TABLE IF EXISTS programari_activitati_grupe;
CREATE TABLE programari_activitati_grupe (
  id_activitate_grupa INT NOT NULL,
  id_grupa INT NOT NULL,
  nr_minim INT NOT NULL DEFAULT 5,
  numar_curent_participanti INT NOT NULL DEFAULT 0,
  data_activitate_grupa DATE,
  ora_activitate INT,
  durata_activitate INT,
  start_programare DATETIME NOT NULL,
  deadline DATETIME NOT NULL,
  PRIMARY KEY (id_activitate_grupa),
  INDEX id_grupa_idx (id_grupa ASC) VISIBLE,
  CONSTRAINT id_grupa
    FOREIGN KEY (id_grupa)
    REFERENCES aplicatie.grupe (id_grupa)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);

DROP TABLE IF EXISTS studenti_grupe;
CREATE TABLE studenti_grupe (
  id_student INT UNSIGNED NOT NULL,
  id_grupa INT NOT NULL,
  INDEX id_stud_idx (id_student ASC) VISIBLE,
  INDEX id_gr_idx (id_grupa ASC) VISIBLE,
  CONSTRAINT id_stud
    FOREIGN KEY (id_student)
    REFERENCES aplicatie.studenti (id_student)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT id_gr
    FOREIGN KEY (id_grupa)
    REFERENCES aplicatie.grupe (id_grupa)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);

DROP TABLE IF EXISTS mesaje_grupe;
CREATE TABLE aplicatie.mesaje_grupe (
  id_from INT UNSIGNED NOT NULL,
  id_grupa INT NOT NULL,
  subiect INT NOT NULL,
  text INT NULL,
  timp_primire TIMESTAMP NOT NULL DEFAULT NOW(),
  programare TINYINT NOT NULL DEFAULT 0,
  PRIMARY KEY (id_from),
  INDEX id_gr_msg_idx (id_grupa ASC) VISIBLE,
  CONSTRAINT id_gr_msg

    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT id_from
    FOREIGN KEY (id_from)
    REFERENCES studenti (id_student)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);

DROP TABLE IF EXISTS mesaje;
CREATE TABLE mesaje (
  id_from INT UNSIGNED NOT NULL,
  id_to INT UNSIGNED NOT NULL,
  subiect VARCHAR(32) NOT NULL,
  mesaj TEXT NULL,
  timp_primire TIMESTAMP NOT NULL DEFAULT NOW(),
  INDEX id_from_msg_idx (id_from ASC) VISIBLE,
  INDEX id_to_msg_idx (id_to ASC) VISIBLE,
  CONSTRAINT id_from_msg
    FOREIGN KEY (id_from)
    REFERENCES users (id_user)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT id_to_msg
    FOREIGN KEY (id_to)
    REFERENCES users (id_user)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);

DROP TABLE IF EXISTS examinari;
CREATE TABLE examinari (
  id_examinari INT NOT NULL,
  id_tip_examinare INT UNSIGNED NOT NULL,
  data_desfasurare DATETIME NOT NULL,
  durata INT NOT NULL DEFAULT 1,
  id_activ_curs INT UNSIGNED NOT NULL,
  PRIMARY KEY (id_examinari),
  INDEX tip_examinare_idx (id_tip_examinare ASC) VISIBLE,
  INDEX activitate_curs_idx (id_activ_curs ASC) VISIBLE,
  CONSTRAINT tip_examinare
    FOREIGN KEY (id_tip_examinare)
    REFERENCES activitati (id_activitate)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT activitate_curs
    FOREIGN KEY (id_activ_curs)
    REFERENCES curs_activitati (id_ca)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);

DROP VIEW IF EXISTS vStudenti;
CREATE VIEW vStudenti AS
SELECT u.id_user, s.id_student, u.CNP, r.descriere_rol, u.nume, u.prenume, u.adresa, u.nr_tel, u.email, u.IBAN, u.nr_contract, s.an_studiu, s.nr_ore, sa.descriere
    FROM users u JOIN studenti s JOIN roluri r JOIN status_activ sa ON u.id_user = s.id_student AND u.id_rol = r.id_rol AND sa.id_status = s.id_status;



SELECT * FROM vStudenti;



DROP VIEW IF EXISTS vProfesori;
CREATE VIEW vProfesori AS
SELECT profesori.id_profesor, CNP, descriere_rol, nume, prenume, adresa, nr_tel, email, IBAN, nr_contract, nr_min_ore, nr_max_ore, cod_departament, descriere_departament, profesori.id_departament
    FROM users JOIN profesori JOIN roluri JOIN departamente ON users.id_user = profesori.id_profesor AND roluri.id_rol = users.id_rol AND departamente.id_departament = profesori.id_departament;



SELECT * FROM vProfesori;



DROP VIEW IF EXISTS vNote;
CREATE VIEW vNote AS
SELECT ca.id_ca, ca.id_curs, s.id_student as 'id_student', sa.nota as 'nota', c.denumire as 'denumire_materie', ca.procent as 'procent', a.denumire_activitate as 'tip_activitate', u.nume as 'nume_prof', u.prenume as 'prenume_prof'
FROM studenti_activitati sa JOIN curs_activitati ca JOIN activitati a JOIN studenti s JOIN cursuri c JOIN users u
ON s.id_student = sa.id_student AND sa.id_activitate = a.id_activitate AND c.id_curs = ca.id_curs AND id_user = ca.id_prof_titular;
-- util pentru notari daca ii dam cu un WHERE id_ca sa coincida cu cele ale profesorului de curs, ii pot filtra dupa curs si sa le dea note
-- se mai poate folosi si la grupuri, cautam studentii inscrisi la denumire_materie = ... si ii sugeram pentru grupul format
SELECT * FROM vNote;



-- view amplu al cursului
DROP VIEW IF EXISTS vCursuri;
CREATE VIEW vCursuri AS
SELECT c.id_curs, ca.id_ca, c.denumire as 'denumire_curs', c.descriere as 'descriere_curs', ca.nr_zi_sapt, ca.ora_inceput, a.nr_max_participanti, u.nume, u.prenume
FROM curs_activitati ca JOIN activitati a JOIN users u JOIN cursuri c
ON ca.id_curs = c.id_curs AND ca.id_activ = a.id_activitate AND ca.id_prof_titular = u.id_user
WHERE a.id_activitate = 1; -- cautam dupa cursuri



SELECT * FROM vCursuri;

-- folosit pentru a asigna studentii la cursuri, seminarii si laboratoare
DROP VIEW IF EXISTS vProfNrCursanti;
CREATE VIEW vProfNrCursanti AS
SELECT ca.id_ca, c.id_curs, ca.id_prof_titular, c.denumire, c.descriere, count(ca.id_ca) as 'nr_actual_participanti', a.nr_max_participanti as 'numar_maxim', ca.nr_zi_sapt as 'ziua', ca.ora_inceput, ca.durata, ca.perioada
FROM curs_activitati ca JOIN studenti_activitati sa JOIN activitati a JOIN cursuri c
ON ca.id_ca = sa.id_activitate AND ca.id_activ = a.id_activitate AND ca.id_curs = c.id_curs -- sa nu bagam la laboratoare mai mult de 15 studenti
-- WHERE daca_mai_este_necesar, daca este asignat la cursul lui X, seminarul lui X dar nu mai gaseste loc la lab lui X, atunci schimbam la cursul lui Y si facem iar verificari de program
GROUP BY (ca.id_ca)
HAVING count(ca.id_ca) < a.nr_max_participanti;
-- Salutare, ai fost asignat la activitatea ... din data (deoarece ti se potriveste), la profesorul X (care are cei mai putini studenti si program perfect pentru tine)
-- sau... Salutare, nu ai putut fi asignat la nicio activitate ...
-- exista sansa sa gaseasca laborator in saptamana para iar in saptamana impara sa faca altceva
SELECT * FROM vProfNrCursanti;

DELIMITER %%
CREATE PROCEDURE pInsertAdmin(IN p_id_rol INT, IN p_cnp CHAR(13), IN p_nume VARCHAR(64), IN p_prenume VARCHAR(64), IN p_adresa VARCHAR(128), IN p_nr_tel CHAR(10), IN p_email VARCHAR(64), IN p_iban VARCHAR(34), IN p_nr_contract INT)
BEGIN
	SET @username = concat(p_nume, '.', p_prenume, SUBSTR(p_cnp, 3, 3));
    SET @parola = concat(p_nume, SUBSTR(p_cnp, 5, 3));
	INSERT INTO users (id_rol, cnp, nume, prenume, adresa, nr_tel, email, iban, nr_contract, username, parola) VALUES (p_id_rol, p_cnp, p_nume, p_prenume, p_adresa, p_nr_tel, p_email, p_iban, p_nr_contract, @username, @parola);
END %%
DELIMITER ;

drop procedure if exists profesorCuStudMin;

delimiter //
create procedure profesorCuStudMin(in idCurs int)
begin
    select id_prof_titular, count(id_student) nrStud
    from curs_activitati ca
             join studenti_activitati sa on ca.id_ca = sa.id_activitate
    where id_curs = idCurs
    group by id_prof_titular
    order by nrStud
    limit 1;
end//

drop procedure if exists getActivitatiCurs;
delimiter //
create procedure getActivitatiCurs(idCurs int, idStudent int, idProfTitular int, idActivitate int)
begin
    select *
    from curs_activitati ca
    where ca.id_curs = idCurs
      and ca.id_prof_titular = idProfTitular
      and idActivitate = ca.id_activ
      and (ca.perioada, ca.nr_zi_sapt, ca.ora_inceput) not in
          (select perioada, nr_zi_sapt, ora_inceput from vActivitatiStudent where id_student = idStudent)
    union
    select *
    from curs_activitati ca
    where ca.id_curs = idCurs
      and ca.id_prof_titular = idProfTitular
      and idActivitate = ca.id_activ
      and (ca.perioada, ca.nr_zi_sapt) in (select perioada, nr_zi_sapt
                                           from vActivitatiStudent
                                           where id_student = idStudent
                                               and (ca.ora_inceput < vActivitatiStudent.ora_inceput and
                                                    vActivitatiStudent.ora_inceput - ca.ora_inceput >= ca.durata)
                                              or (ca.ora_inceput > vActivitatiStudent.ora_inceput and
                                                  ca.ora_inceput - vActivitatiStudent.ora_inceput >=
                                                  vActivitatiStudent.durata))
    union
    select *
    from curs_activitati ca
    where ca.id_curs = idCurs
      and ca.id_prof_titular = idProfTitular
      and idActivitate = ca.id_activ
      and (ca.perioada, ca.nr_zi_sapt, ca.ora_inceput) in (select perioada, nr_zi_sapt, ora_inceput
                                                           from vActivitatiStudent
                                                           where id_student = idStudent
                                                             and ((ca.perioada = 0 and perioada = 1) or
                                                                  ca.perioada = 1 and perioada = 0));
end//

drop procedure if exists renuntaCurs;

delimiter //
create procedure renuntaCurs(id_student int, descriereCurs varchar(128))
begin
	select id_curs into @idCurs from cursuri where descriere = descriereCurs;

    delete
    from studenti_activitati
    where id_activitate in (select id_ca from curs_activitati where id_curs = @idCurs)
      and id_student = id_student;
end//

drop procedure if exists calendar;

delimiter //
create procedure calendar(idStud int)
begin
    select data_inceput into @dataInceput from an_universitar;
    select floor(datediff(current_date(), @dataInceput) / 7) % 2 into @tipActivitate;
    select c.descriere                                  as "Disciplina",
           a.denumire_activitate                        as "Activitate",
           p.descriere                                  as "Perioada",
           vas.nr_zi_sapt                               as "Zi",
           vas.ora_inceput                              as "Ora de inceput",
           vas.durata                                   as "Durata",
           concat(prof.nume, concat(' ', prof.prenume)) as "Nume Profesor"
    from vactivitatistudent vas
             join cursuri c
             join users prof
             join perioade p
             join activitati a
                  on vas.id_prof_delegat = prof.id_user and vas.id_curs = c.id_curs and vas.perioada = p.id_perioada and
                     vas.id_activ = a.id_activitate
    where id_student = idStud
      and (vas.perioada >= 2 or perioada = @tipActivitate)
    order by Zi, vas.ora_inceput;
end //

drop procedure if exists notePartiale;

delimiter //
create procedure notePartiale(idStud int)
begin
    select c.descriere                            as "Disciplina",
           a.denumire_activitate                  as "Activitate",
           concat(u.nume, concat(' ', u.prenume)) as "Profesor",
           sa.data_notare                         as "Data",
           sa.nota                                as "Nota",
           ca.procent                             as "Procent"
    from curs_activitati ca
             join users u
             join activitati a
             join cursuri c
             join studenti_activitati sa
                  on ca.id_ca = sa.id_activitate and ca.id_curs = c.id_curs and ca.id_prof_delegat = u.id_user and
                     a.id_activitate = ca.id_activ
    where id_student = idStud;
end//



drop procedure if exists noteFinale;

delimiter //
create procedure noteFinale(idStud int)
begin
    select distinct c.descriere as "Disciplina",
                    concat(u.nume, concat(' ', u.prenume)) as "Profesor",
                    truncate(sum(nota * (procent / 100)), 2) as "Nota"
    from curs_activitati ca
             join cursuri c
             join users u
             join studenti_activitati sa
                  on ca.id_ca = sa.id_activitate and c.id_curs = ca.id_curs and u.id_user = ca.id_prof_titular
    where sa.id_student = idStud
    group by c.descriere, concat(u.nume, concat(' ', u.prenume))
    having count(sa.id_activitate) = count(sa.nota);
end//



drop procedure if exists pDeleteActivitate;

delimiter //
CREATE DEFINER=`root`@`localhost` PROCEDURE `pDeleteActivitate`(curs int,activitate int)
BEGIN
    delete from curs_activitati where id_curs = curs and id_activ = activitate;
    delete from studenti_activitati where id_activitate in (select id_ca from curs_activitati where id_curs = curs and id_activ = activitate);
END//

drop procedure if exists pInsertActivitate;

delimiter //
CREATE DEFINER=`root`@`localhost` PROCEDURE `pInsertActivitate`(id_curs int, id_activ int, perioada int, nr_zi_sapt int, procent int,id_prof_titular int,id_prof_delegat int,ora int,durata int,max_participanti int)
BEGIN
    set @new_id = if ((select count(*) from curs_activitati)>0,(select max(id_ca) from curs_activitati),0);
    insert into curs_activitati value (@new_id+1,id_curs,id_activ,perioada,nr_zi_sapt,procent,id_prof_titular,id_prof_delegat,ora,durata,max_participanti);
END//
